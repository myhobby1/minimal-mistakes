<1부터 10까지의 합>

#include <stdio.h>

 int main(void)
 {
    int i ;
    int sum = 0 ;
    
    for( i=1 ;  i<11 ;  i++)
        sum += i ;

    printf( " 1부터 10까지의 합 ; %d " , sum) ;
   
    return 0 ;
 }
