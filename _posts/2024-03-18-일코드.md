#include <stdio.h>
int main(void)
 {
    int a=10, b=10 ;
    int bef, aft ;
    bef=(++a) * 5 ;
    aft =(b++) * 5 ;
    
    printf(" a=%d, b=%d\n" , a, b) ;
    printf(" (++a) * 5=%d, (b++) * 5=%d\n" , bef, aft) ;

    return 0 ;
 }
